[[http://dotsies.org/][Dotsies]] 是使用点代替常规字母书写的一种拉丁字母显示体系，每个字母由五个点的显隐来
表示。
+ 想法一：制作坚排显示的字体。
+ 想法二：改进，添加定位标识，用于图形识别。

#+begin_html
<!-- link href='http://dotsies.org/dotsies.css' rel='stylesheet' type='text/css' -->
<script type="text/javascript">
function dotsies_hover() {
  var l = document.createElement('link');
  document.getElementsByTagName('head')[0].appendChild(l);
  l.setAttribute('href', 'http://dotsies.org/dotsies.css');
  l.setAttribute('type', 'text/css');
  l.setAttribute('rel', 'stylesheet');
  var s = document.createElement('style');
  document.getElementsByTagName('head')[0].appendChild(s);
  s.innerHTML = ('body, p, div, span, a, li, textarea, pre, input {font-family: Dotsies !important}');
}
</script>
